1.
Set is an associative container.
Only stores key.
One key only appears once.

#include <set>

//create set
set<string> words; //data_type是string
string word = "";
cin >> word;
words.insert(word); //insert的时候自动按照字母顺序排序

set<string>::iterator it = words.begin(); //iterator
cout << *it << endl; //*it 取iterator的内容
